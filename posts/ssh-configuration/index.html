SSH Configuration

Mar 15, 2019
1 min read

Client

# generate public / private keys
$ ssh-keygen -b 4096 -t rsa -C comments -f ~/.ssh/abc
# copy public key to server
$ ssh-copy-id -i ~/.ssh/abc.pub root@192.168.1.200
# ssh-agent && ssh-add
$ ssh-add ~/.ssh/abc
# ssh-add -l check
$ ssh-add -l

Server

# add user A
[root@hostname ~] # adduser A
# set password for user A
[root@hostname ~] # passwd A
# add user A to group `wheel`
[root@hostname ~] # gpasswd -a A wheel

Login

# Login with ssh -o PubkeyAuthentication=yes root@192.168.1.200
$ ssh root@192.168.1.200
